/**
 * Calculates the sum of two numbers.
 * @param {number} a - The first number.
 * @param {number} b - The second number.
 * @returns {number} - The sum of the two numbers.
 */
function add(a, b) {
  return a + b;
}

 * Greet a person.

function greet(name) {
  return `Hello, ${name}!`;
}
